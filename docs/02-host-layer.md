# 02 — Host Layer

The Host Layer is the HTML/CSS application shell responsible for:
- Rendering the game **launcher dashboard** with live-preview game tiles
- Managing **screen transitions** (launcher ↔ game ↔ settings)
- Displaying **system-level UI** (loading spinner, exit button, on-screen controller)
- Reading the game catalogue and building the **live preview tile grid**
- Hosting the **settings panel** accessible from the header

The Host Layer never touches PixiJS or game logic directly — it communicates with the Runtime Layer by showing/hiding the canvas and calling the loader.

---

## Responsibilities

```
Host Layer
├── App Shell Bootstrap (app.js)
│   ├── Register Service Worker
│   ├── Load settings from IndexedDB
│   ├── Load game manifest (games/manifest.json)
│   └── Render launcher UI into #app-shell
│
├── Launcher Dashboard (HTML generated by JS)
│   ├── Header bar (ToyBox logo + ⚙️ settings button)
│   └── Game grid (one tile per game: live preview canvas + title)
│
├── Screen Router
│   ├── showLauncher()    → stop previews, hide canvas, show #app-shell
│   ├── showGame()        → hide #app-shell, show canvas, boot game
│   ├── openSettings()    → slide in settings panel over launcher
│   └── closeSettings()   → slide out settings panel, resume previews
│
└── System Overlay (visible during gameplay)
    ├── Exit game button (z-index 9999, always on top)
    └── Loading spinner (during game boot)
```

## Step 2 — Host Bootstrap (`app.js`)

The host bootstrap runs on page load. It:
1. Registers the Service Worker
2. Loads the game manifest
3. Renders the launcher UI
4. Wires up the exit button

```javascript
// app.js
import { loadGame, exitGame } from './engine/loader.js';

// ── Service Worker Registration ──────────────────────────────────────────────
if ('serviceWorker' in navigator) {
  navigator.serviceWorker.register('/sw.js').catch(console.warn);
}

// ── DOM References ────────────────────────────────────────────────────────────
const appShell       = document.getElementById('app-shell');
const systemOverlay  = document.getElementById('system-overlay');
const btnExitGame    = document.getElementById('btn-exit-game');
const gameCanvas     = document.getElementById('game-canvas');

// ── Screen Router ─────────────────────────────────────────────────────────────
export function showLauncher() {
  appShell.classList.remove('hidden');
  systemOverlay.classList.add('hidden');
  gameCanvas.style.display = 'none';
}

export function showGame() {
  appShell.classList.add('hidden');
  systemOverlay.classList.remove('hidden');
  gameCanvas.style.display = 'block';
}

// ── Exit Button ───────────────────────────────────────────────────────────────
btnExitGame.addEventListener('pointerdown', () => {
  exitGame();
  showLauncher();
});

// ── Launcher Render ───────────────────────────────────────────────────────────
async function renderLauncher() {
  // Load game catalogue (bundled + sideloaded)
  const games = await getAllGames();

  // Apply age filter from settings
  const ageFilter = getSetting('ageFilter');
  const filtered  = ageFilter === 'all'
    ? games
    : games.filter(g => !g.ageRange || g.ageRange === ageFilter);

  // Build HTML structure
  appShell.innerHTML = `
    <header class="launcher-header">
      <span class="launcher-logo">🎮 ToyBox</span>
      <button id="btn-settings" aria-label="Settings">⚙️</button>
    </header>
    <main class="game-grid" id="game-grid">
      ${filtered.map(buildGameTile).join('')}
    </main>
  `;

  // Wire up tile tap handler
  document.getElementById('game-grid').addEventListener('pointerdown', (e) => {
    const tile = e.target.closest('.game-tile');
    if (!tile) return;
    const gameId = tile.dataset.gameId;
    const game   = filtered.find(g => g.id === gameId);
    if (game) onGameSelected(game);
  });

  // Wire up settings button
  document.getElementById('btn-settings').addEventListener('pointerdown', () => {
    // If a parental PIN is set, verify it before opening settings
    const pin = getSetting('parentalPin');
    if (pin) {
      showPinEntry(() => openSettingsPanel());
    } else {
      openSettingsPanel();
    }
  });

  // Start live previews staggered after render
  // See doc 13 (Game Preview System) for full preview implementation
  requestAnimationFrame(() => initAllPreviews(filtered));
}

// ── Game Tile Builder ────────────────────────────────────────────────────────────────────────────────
// Tile layout: live preview canvas (top) + title footer (bottom).
// See doc 13 (Game Preview System) for the full preview subsystem.
function buildGameTile(game) {
  const PREVIEW_W = 240;
  const PREVIEW_H = 150; // 8:5 ratio
  return `
    <div class="game-tile" data-game-id="${game.id}" role="button" tabindex="0"
         aria-label="Play ${game.title}">

      <div class="game-tile__preview-wrap">
        <canvas class="game-tile__preview-canvas"
                id="preview-canvas-${game.id}"
                width="${PREVIEW_W}" height="${PREVIEW_H}"
                aria-hidden="true"></canvas>
        <div class="game-tile__play-overlay" aria-hidden="true">
          <span class="game-tile__play-icon">▶</span>
        </div>
      </div>

      <div class="game-tile__footer">
        <span class="game-tile__title">${game.title}</span>
        ${game.sideloaded ? '<span class="game-tile__badge">Custom</span>' : ''}
        ${game.config?.controller?.enabled
          ? '<span class="game-tile__badge game-tile__badge--ctrl">🎮</span>'
          : ''}
      </div>

    </div>
  `;
}

// ── Game Selection Handler ────────────────────────────────────────────────────
async function onGameSelected(game) {
  showLoadingScreen();
  try {
    await loadGame(game, { onReady: () => { hideLoadingScreen(); showGame(); } });
  } catch (err) {
    console.error('[ToyBox] Failed to load game:', err);
    hideLoadingScreen();
    showLauncher();
  }
}

// ── Loading Screen Helpers ────────────────────────────────────────────────────
function showLoadingScreen() {
  const screen = document.createElement('div');
  screen.id = 'loading-screen';
  screen.className = 'loading-screen';
  screen.innerHTML = `<div class="spinner"></div><p>Loading...</p>`;
  document.body.appendChild(screen);
}

function hideLoadingScreen() {
  document.getElementById('loading-screen')?.remove();
}

// ── Init ──────────────────────────────────────────────────────────────────────
renderLauncher();
```

## Step 3 — Screen Transition Strategy

| Transition | Method | Duration |
|-----------|--------|----------|
| Launcher → Game | Stop previews, fade `#app-shell` to opacity 0 | 300ms |
| Game → Launcher | Fade `#app-shell` to opacity 1, restart previews | 300ms |
| Launcher → Settings | Slide settings panel in from right | 350ms |
| Settings → Launcher | Slide settings panel out to right | 350ms |
| Loading overlay | Injected DOM element with spinner | Until game ready |

## Step 4 — Responsive Grid Layout

The game grid uses CSS `auto-fill` so it automatically adapts to the tablet's screen width. With the new wider preview canvas tiles (240px minimum), the columns adjust accordingly:

| Viewport Width | Columns |
|---------------|---------|
| < 600px | 2 columns |
| 600–960px | 3 columns |
| 960–1280px | 4 columns |
| > 1280px | 5 columns |

No media queries needed — `grid-template-columns: repeat(auto-fill, minmax(240px, 1fr))` handles this.

## Step 5 — Settings Navigation

The `⚙️` button in the launcher header opens the Settings panel. Full implementation is covered in [12 — Settings System](./12-settings-system.md). Key host-layer responsibilities:

1. **PIN Gate**: If `parentalPin` is set in settings, show the PIN entry overlay before opening the settings panel
2. **Stop Previews**: Call `stopAllPreviewers()` before opening settings to free canvas resources
3. **Resume Previews**: Restart preview animations when the settings panel closes
4. **Live Updates**: `onSettingChange('ageFilter', ...)` re-renders the game grid immediately

```javascript
// app.js — settings button handler
document.getElementById('btn-settings').addEventListener('pointerdown', () => {
  stopAllPreviewers(); // Pause previews while settings are open
  const pin = getSetting('parentalPin');
  if (pin) {
    showPinEntry(() => openSettingsPanel());
  } else {
    openSettingsPanel();
  }
});

// When settings close, restart previews
onSettingChange('*', () => {
  // Re-render the grid if age filter changed
});
```

## Step 6 — Accessibility Considerations

Even though this is a kids' app, basic accessibility makes the app work better across devices:

- All interactive elements have `aria-label` attributes
- Game tiles have `role="button"` and `tabindex="0"` for keyboard navigation (useful during development)
- The exit button is 56×56px minimum — larger than the 44px touch target minimum per Apple HIG
- Color contrast ratios meet WCAG AA for text on backgrounds
- Settings panel uses `role="dialog"` and `aria-modal="true"` for screen reader support
- D-pad buttons have individual `aria-label` attributes
